What are the percentages or late flights(departure and arrival) per carrier?

carrier_list <- airlines$carrier

A function that given a list of carrier names will calculate their personal percentage of late departures and late arrivals and insert the information into a data frame

function1 <- function(carrier_list){
  delay <- c()
  for (i in 1:length(carrier_list)) {
    x <- flights%>%
      filter(carrier == carrier_list[i])%>%
      nrow()
    y <- flights %>%
      filter(carrier == carrier_list[i] & dep_delay > 0)%>%
      nrow()
    z <- (y/x)*100
    delay <- append(delay, z)
  }
  return(delay)
}

function2 <- function(carrier_list){
  delay <- c()
  for (i in 1:length(carrier_list)) {
    x <- flights%>%
      filter(carrier == carrier_list[i])%>%
      nrow()
    y <- flights %>%
      filter(carrier == carrier_list[i] & arr_delay > 0)%>%
      nrow()
    z <- (y/x)*100
    delay <- append(delay, z)
  }
  return(delay)
}
departure_delay <- function1(carrier_list)
arrival_delay <- function2(carrier_list)
late_carrier <- data.frame(carrier_list,departure_delay, arrival_delay)

What is the proportion of airtime afforded to each airline/carrier?

function4 <- function(carrier_list){
  airtime_percentage <- c()
  total_airtime <- flights%>%
    dplyr::select(air_time)%>%
    sum(na.rm = TRUE)
  for(i in carrier_list){
    carrier_airtime <- flights%>%
      filter(carrier == i)%>%
      dplyr::select(air_time)%>%
      sum(na.rm = TRUE)
    z <- (carrier_airtime/total_airtime)*100
    airtime_percentage <- append(airtime_percentage,z)
  }
  return (airtime_percentage)
}
carrier_air <- function4(carrier_list)
carrier_airtime <- data.frame(carrier_list, carrier_air)
